// The metadata preamble renders as a card; these lines are then STRIPPED from
// the preamble markdown so they don't render twice. Anything else in the
// preamble (rare legacy prose) still renders below the card. The title line
// is stripped too — the card carries the version.
const METADATA_LINE_RE =
  /^(?:# .*|Component:.*|Master plan:.*|Date:.*|Provenance:.*|Supersedes:.*)$/;
export function stripPreambleMetadata(body: string): string {
  return body
    .split("\n")
    .filter((ln) => !METADATA_LINE_RE.test(ln.trim()))
    .join("\n")
    .trim();
}

function MetadataCard({ parsed }: { parsed: ParsedExecutionPlan }) {
  const rows: Array<[string, React.ReactNode]> = [];

  if (parsed.date) rows.push(["Date", parsed.date]);
  if (parsed.provenance) rows.push(["Provenance", parsed.provenance]);
  if (parsed.supersedes) rows.push(["Supersedes", parsed.supersedes]);
  if (parsed.masterPlan) rows.push(["Master plan", parsed.masterPlan]);
  if (rows.length === 0) return null;
  return (
    <div className="mb-4 grid grid-cols-[auto_1fr] gap-x-4 gap-y-1 rounded-lg border border-stone-200 bg-stone-50 p-4 text-sm dark:border-stone-700 dark:bg-stone-800/50">

        <React.Fragment key={k}>
          <span className="font-medium" style={{ color: "var(--rp-prose-muted)" }}>{k}</span>
          <span>{v}</span>
        </React.Fragment>
      ))}
    </div>
  );
}

```tsx
// Build steps render as a numbered card spine (spec R2). Boundaries come from
// marked's lexer (structural, never regex over raw text); each item's inner
// markdown re-renders through BodyParts so agent-detail blocks keep working.
// Cards add label text nodes but never remove body text — AnnotationLayer's
// offset anchoring stays valid for step content.
import { Marked } from "marked";
const stepLexer = new Marked({ gfm: true });

function splitBuildSteps(body: string): { before: string; items: string[]; after: string } | null {
  const tokens = stepLexer.lexer(body);
  const listIdx = tokens.findIndex((t) => t.type === "list" && (t as { ordered?: boolean }).ordered);
  if (listIdx === -1) return null;
  const list = tokens[listIdx] as { raw: string; items: Array<{ text: string }> };
  const start = body.indexOf(list.raw);
  if (start === -1) return null; // lexer normalized the raw — fall back to plain rendering
  return {
    before: body.slice(0, start).trim(),
    items: list.items.map((it) => it.text),
    after: body.slice(start + list.raw.length).trim(),
  };
}

function StepCards({ body, detailOpen }: { body: string; detailOpen: boolean }) {
  const split = useMemo(() => splitBuildSteps(body), [body]);
  if (!split) return <BodyParts source={body} detailOpen={detailOpen} />;
  return (
    <>
      {split.before && <BodyParts source={split.before} detailOpen={detailOpen} />}
      {split.items.map((text, i) => (
        <div
          key={i}
          className="my-3 rounded-lg border border-stone-200 p-4 dark:border-stone-700"
        >
          <div className="mb-1 text-[11px] font-semibold uppercase tracking-wide" style={{ color: "var(--rp-prose-muted)" }}>
            Step {i + 1} of {split.items.length}
          </div>
          <BodyParts source={text} detailOpen={detailOpen} />
        </div>
      ))}
      {split.after && <BodyParts source={split.after} detailOpen={detailOpen} />}
    </>
  );
}
```

```tsx
// Scroll-spy for the sidebar outline (spec R2): observe the reading
// container's headings; the active one is the last heading whose top passed
// the upper reading band. Returns the ELEMENT — callers map it to their
// outline-entry id. No-ops when IntersectionObserver is unavailable.
import { useEffect, useState, type RefObject } from "react";

export function useScrollSpy(
  ref: RefObject<HTMLElement | null>,
  deps: unknown[],
): Element | null {
  const [active, setActive] = useState<Element | null>(null);
  useEffect(() => {
    const host = ref.current;
    if (!host || typeof IntersectionObserver === "undefined") return;
    const headings = Array.from(host.querySelectorAll("h1, h2, h3"));
    if (headings.length === 0) return;
    const seen = new Map<Element, boolean>();
    const io = new IntersectionObserver(
      (entries) => {
        for (const e of entries) seen.set(e.target, e.isIntersecting || e.boundingClientRect.top < 0);
        const passed = headings.filter((h) => seen.get(h));
        setActive(passed.length ? passed[passed.length - 1] : null);
      },
      { rootMargin: "0px 0px -70% 0px" },
    );
    headings.forEach((h) => io.observe(h));
    return () => io.disconnect();
    // eslint-disable-next-line react-hooks/exhaustive-deps
  }, deps);
  return active;
}
```
